  <!-- Buttons for Different Scripts -->
  <div class="button-group">
    <button onclick="generateGetAdUser()">Get AD User</button>
    <button onclick="generatePasswordReset()">Password Last Set</button>
    <button onclick="generateSearchBySID()">Search by SID</button>

  <!-- Output Box for the Generated Script -->
  <div class="output-box" id="scriptOutput">
    <!-- Default placeholder -->
    !setIncident ...
  </div>

  <script>
    // Function to generate a basic Get-ADUser command
    function generateGetAdUser() {
      const username = document.getElementById("username").value.trim();
      if (!username) {
        alert("Please enter a username.");
        return;
      }
      const script = `Get-ADUser -Identity "${username}"`;
      document.getElementById("scriptOutput").textContent = script;
    }

    // Function to generate a command to get the password last set property
    function generatePasswordReset() {
      const username = document.getElementById("username").value.trim();
      if (!username) {
        alert("Please enter a username.");
        return;
      }
      const script = `Get-ADUser -Identity "${username}" -Properties PasswordLastSet | Select-Object Name,PasswordLastSet`;
      document.getElementById("scriptOutput").textContent = script;
    }

    // Function to generate a command to search by SID
    function generateSearchBySID() {
      const sid = document.getElementById("sid").value.trim();
      if (!sid) {
        alert("Please enter a SID for this command, or use the AD User search if you don't have a SID.");
        return;
      }
      const script = `Get-ADUser -Filter { SID -eq '${sid}' }`;
      document.getElementById("scriptOutput").textContent = script;
    }
  </script>
